def make_venn_subsets(a,b,c):
    return (len(a-(b.union(c))), 
    len(b - (a.union(c))), 
    len(a.intersection(b) - c), 
    len(c - (a.union(b))), 
    len(a.intersection(c) - b), 
    len(b.intersection(c) - a), 
    len(a.intersection(b).intersection(c)))
